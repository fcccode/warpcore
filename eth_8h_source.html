97 eth_tx(struct w_engine * const w, struct w_iov * const v, const uint16_t len)
98 {
99  // find a tx ring with space
100  struct netmap_ring * txr = 0;
101  for (uint32_t r = 0; likely(r < w->b->nif->ni_tx_rings); r++) {
102  txr = NETMAP_TXRING(w->b->nif, w->b->cur_txr);
103  if (likely(!nm_ring_empty(txr)))
104  // we have space in this ring
105  break;
106 
107  warn(INF, "tx ring %u full; moving to next", w->b->cur_txr);
108  w->b->cur_txr = (w->b->cur_txr + 1) % w->b->nif->ni_tx_rings;
109  txr = 0;
110  }
111 
112  // return false if all rings are full
113  if (unlikely(txr == 0)) {
114  warn(INF, "all tx rings are full");
115  return false;
116  }
117 
118  struct netmap_slot * const s = &txr->slot[txr->cur];
119  w->b->slot_buf[txr->ringid][txr->cur] = v;
120  s->len = len + sizeof(struct eth_hdr);
121 
122  if (unlikely(nm_ring_space(txr) == 1 || sq_next(v, next) == 0)) {
123  // we are using the last slot in this ring, or this is the last w_iov in
124  // this batch - mark the slot for reporting
125  s->flags = NS_REPORT | NS_BUF_CHANGED;
126  } else
127  s->flags = NS_BUF_CHANGED;
128 
129  warn(DBG, "%s iov idx %u into tx ring %u slot %d (%s %u)",
130  is_pipe(w) ? "copying" : "placing", v->idx, w->b->cur_txr, txr->cur,
131  is_pipe(w) ? "idx" : "swap with", s->buf_idx);
132  if (unlikely(is_pipe(w)))
133  // for netmap pipes, we need to copy the buffer into the slot
134  memcpy(NETMAP_BUF(txr, s->buf_idx), IDX2BUF(w, v->idx), s->len);
135  else {
136  // for NIC rings, temporarily place v into the current tx ring
137  const uint32_t slot_idx = s->buf_idx;
138  s->buf_idx = v->idx;
139  v->idx = slot_idx;
140  }
141 
142 #if !defined(NDEBUG) && DLEVEL >= DBG
143  char src[ETH_ADDR_STRLEN];
144  char dst[ETH_ADDR_STRLEN];
145  const struct eth_hdr * const eth = (void *)NETMAP_BUF(txr, s->buf_idx);
146  warn(DBG, "Eth %s -> %s, type %d, len %lu", ether_ntoa_r(&eth->src, src),
147  ether_ntoa_r(&eth->dst, dst), ntohs(eth->type), len + sizeof(*eth));
148 #endif
149

151  txr->head = txr->cur = nm_ring_next(txr, txr->cur);
